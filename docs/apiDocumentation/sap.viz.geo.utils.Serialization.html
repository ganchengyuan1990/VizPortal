[static] Serialization.updateDataset (value, datasetArr)

Update all geomap layer's dataset, it will change input value directly. if no datasetArr, just remove all layers' data.
Parameters:
  • value [Optional]
    • serialization result from sap.viz.geo.Map.toJSON()
  • datasetArr [Optional]
    • array of Flattable to update value
Returns:
    • updated serialization result

Example

How to use geo serialization utils

var chart = sap.viz.api.core.createViz({ type : 'viz/geomap',
         container : $('#mapContainer'), options : { geomap : { baseMap :
         baseMapProvider } } });

var map = chart.action("getMap");
var data = { "metadata" : { "fields" : [
          { "id" : "Location",
            "name" : "Location",
            "semanticType" : "Dimension",
            "dataType" : "String" },
          { "id" : "Value",
            "name" : "Value",
            "semanticType" : "Measure" }] }, 
          "data" : [["21017506,NAVTEQ", 3000],
                              ["21010593,NAVTEQ", 4000],
                              ["21020341,NAVTEQ", 5000]],
          "info" : [{ "type" : "geo", 
                                "info" : [[{'type' : sap.viz.geo.LocationType.FEATURE_ID }],
                [{ 'type' : sap.viz.geo.LocationType.FEATURE_ID }],
                 [{ 'type' : sap.viz.geo.LocationType.FEATURE_ID }]] }] };

var ds = new sap.viz.api.data.FlatTableDataset(data); var feeding = {
location : 'Location', color : 'Value' };

map.addLayer(new sap.viz.geo.dataviz.Layer({ viz : new
sap.viz.geo.dataviz.Choropleth(), data : ds, feeding : feeding }));

var json = map.toJSON();
var newFeeding = sap.viz.geo.utils.serialization.getFeeds(json);
json = sap.viz.geo.utils.serialization.removeDataset(json);

var updateData = { "metadata" : { "fields" :
         [{ "id" : "Location", 
            "name" : "Location",
            "semanticType" : "Dimension",
            "dataType" : "String" },
          { "id" : "Value", 
            "name" : "Value",
            "semanticType" : "Measure" }] },
          "data" :
                [["21017506,NAVTEQ", 2000],
                ["21010593,NAVTEQ", 3000],
                ["21020341,NAVTEQ",500]],
          "info" : [{ "type" : "geo",
                                "info" : [[{ 'type' : sap.viz.geo.LocationType.FEATURE_ID }], 
          [{ 'type' : sap.viz.geo.LocationType.FEATURE_ID }], 
          [{ 'type' : sap.viz.geo.LocationType.FEATURE_ID }]] }] };

var newData = new sap.viz.api.data.FlatTableDataset(updateData); 
json =  sap.viz.geo.utils.serialization.updateDataset(json, [newData]);
var chart2 = sap.viz.api.core.createViz({ type : 'viz/geomap', container : $('#chart'));
var geomap = chart2.action("getMap"); geomap.fromJSON(json);
